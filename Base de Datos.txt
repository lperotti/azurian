- Script de creación de tabla:
CREATE TABLE `azurian`.`nombres` (
  `id` INT NOT NULL,
  `nombre` VARCHAR(45) NULL,
  `aPaterno` VARCHAR(45) NULL,
  `aMaterno` VARCHAR(45) NULL,
  `edad` INT NULL,
  PRIMARY KEY (`id`));
